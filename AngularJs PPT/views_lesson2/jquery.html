Using AngualrJs with jQuery

.directive( 'myDirective', function () {
		return {
			template: 'Toggle me!',
			link: function ( scope, element, attrs ) {
				var on = false;
				$(element).click( function () {
					if ( on ) {
						$(element).removeClass( 'active' );
					}
					else {
						$(element).addClass( 'active' );
					}
					on = !on;
				});
			}
		};
});

Using Angulars

.directive( 'myDirective', function () {
	return {
		scope: true,
		template: 'Toggle me!',
		link: function ( scope, element, attrs ) {
			scope.on = false;
 
			scope.toggle = function () {
				scope.on = !$scope.on;
			};
		}
	};
});
